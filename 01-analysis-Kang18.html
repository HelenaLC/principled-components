names(dr) <- dr <- c("PCA")

  SIMPLIFY = FALSE,
  dr = df$dr, id = df$id,
  \(dr, id) .pcr(sce, id, dr)) |>
  do.call(what = rbind) |>
  `rownames<-`(NULL)
